    <script>
        // Khởi tạo Lucide Icons
        lucide.createIcons();

        // Xử lý chuyển tab
        const navButtons = document.querySelectorAll('.nav-button');
        const tabContents = document.querySelectorAll('.tab-content');
        const appContainer = document.getElementById('app-container'); // Lấy container chính

        navButtons.forEach(button => {
            button.addEventListener('click', () => {
                const tabId = button.getAttribute('data-tab');

                // Bỏ active khỏi tất cả button và content
                navButtons.forEach(btn => btn.classList.remove('active', 'bg-blue-100', 'text-blue-700'));
                tabContents.forEach(content => content.classList.remove('active'));

                // Thêm active cho button và content được click
                button.classList.add('active', 'bg-blue-100', 'text-blue-700');
                document.getElementById(`tab-${tabId}`).classList.add('active');

                 // Cuộn lên đầu trang khi chuyển tab
                 const mainContentArea = appContainer.querySelector('main');
                 if (mainContentArea) {
                    mainContentArea.scrollTop = 0;
                 }
            });
        });

         // Đảm bảo tab đầu tiên được active khi tải trang
         document.addEventListener('DOMContentLoaded', () => {
            const initialActiveButton = document.querySelector('.nav-button[data-tab="home"]');
             const initialActiveTab = document.getElementById('tab-home');
             if (initialActiveButton && initialActiveTab) {
                 initialActiveButton.classList.add('active', 'bg-blue-100', 'text-blue-700');
                 initialActiveTab.classList.add('active');
             }
         });

    </script>
